

# Create a Postgresql-persistent 
oc new-app --name postgresql-persistent --image registry.redhat.io/rhel8/postgresql-13:1-7  -e POSTGRESQL_USER=redhat -e POSTGRESQL_PASSWORD=redhat123 -e POSTGRESQL_DATABASE=persistentdb

# Set storage
oc set volumes deployment/postgresql-persistent --add --type pvc --name postgresql-storage --claim-sie 10Gi --claim-class nfs-storage --claim-mode rwo --mount-path /var/lib/pgsql --claim-name postgresql-storage
